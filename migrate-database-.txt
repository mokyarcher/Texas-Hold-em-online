# 数据库迁移脚本 - 添加 avatar_filename 字段
# 适用于 Windows PowerShell

$ErrorActionPreference = "Stop"

Write-Host "=========================================" -ForegroundColor Cyan
Write-Host "  数据库迁移 - 添加 avatar_filename 字段" -ForegroundColor Cyan
Write-Host "=========================================" -ForegroundColor Cyan
Write-Host ""

$DB_PATH = "backend\db\poker.db"

# 1. 检查数据库文件
Write-Host "步骤 1/4: 检查数据库文件..." -ForegroundColor Yellow
if (-not (Test-Path $DB_PATH)) {
    Write-Host "✗ 错误: 数据库文件不存在: $DB_PATH" -ForegroundColor Red
    Write-Host ""
    Read-Host "按任意键退出..."
    $null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
    exit 1
}
Write-Host "✓ 数据库文件存在" -ForegroundColor Green
Write-Host ""

# 2. 检查 avatar_filename 字段
Write-Host "步骤 2/4: 检查 avatar_filename 字段..." -ForegroundColor Yellow
$columns = sqlite3 $DB_PATH "PRAGMA table_info(users);"
$avatarExists = $false

foreach ($line in $columns) {
    if ($line -match "avatar_filename") {
        $avatarExists = $true
        break
    }
}

if ($avatarExists) {
    Write-Host "⚠ 警告: avatar_filename 字段已存在，跳过迁移" -ForegroundColor Yellow
    Write-Host ""
    Write-Host "按任意键退出..."
    $null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
    exit 0
}
Write-Host "✓ avatar_filename 字段不存在，需要添加" -ForegroundColor Green
Write-Host ""

# 3. 添加 avatar_filename 字段
Write-Host "步骤 3/4: 添加 avatar_filename 字段..." -ForegroundColor Yellow
$result = sqlite3 $DB_PATH "ALTER TABLE users ADD COLUMN avatar_filename TEXT DEFAULT 'avatar1.png';"

if ($LASTEXITCODE -eq 0) {
    Write-Host "✓ avatar_filename 字段添加成功" -ForegroundColor Green
} else {
    Write-Host "✗ 错误: avatar_filename 字段添加失败" -ForegroundColor Red
    Write-Host ""
    Read-Host "按任意键退出..."
    $null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
    exit 1
}

# 4. 验证字段
Write-Host ""
Write-Host "步骤 4/4: 验证字段..." -ForegroundColor Yellow
$columnsAfter = sqlite3 $DB_PATH "PRAGMA table_info(users);"
$avatarExistsAfter = $false

foreach ($line in $columnsAfter) {
    if ($line -match "avatar_filename") {
        $avatarExistsAfter = $true
        break
    }
}

if ($avatarExistsAfter) {
    Write-Host "✓ avatar_filename 字段验证成功" -ForegroundColor Green
} else {
    Write-Host "✗ 错误: avatar_filename 字段验证失败" -ForegroundColor Red
    Write-Host ""
    Read-Host "按任意键退出..."
    $null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
    exit 1
}

Write-Host ""
Write-Host "=========================================" -ForegroundColor Cyan
Write-Host "  迁移完成" -ForegroundColor Cyan
Write-Host "=========================================" -ForegroundColor Cyan
Write-Host ""
Write-Host "✓ 数据库迁移成功完成！" -ForegroundColor Green
Write-Host ""
Write-Host "现在可以重启应用了：" -ForegroundColor White
Write-Host "  cd backend" -ForegroundColor Gray
Write-Host "  npm start" -ForegroundColor Gray
Write-Host ""
Write-Host "按任意键退出..." -ForegroundColor Yellow
$null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
